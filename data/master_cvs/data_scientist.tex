% --- Project 1 ---
\noindent\textbf{Early Detection of Alzheimer’s Disease Using 3D Swin Transformer} \hfill \textit{MSc Thesis}
% CHANGED: Increased 'topsep' to add space above the list. Increased 'itemsep' for space between points.
\begin{itemize}[leftmargin=*, topsep=4pt, itemsep=2pt]
    \item Developed a novel \textbf{tri-modal deep learning framework} for early Alzheimer’s Disease diagnosis, integrating 3D MRI scans, longitudinal clinical data, and biomarker sequences from the ADNI dataset (N=187).
    \item Designed and pre-trained a \textbf{3D Swin Transformer} with a self-supervised contrastive learning strategy to mitigate data scarcity and model collapse in low-sample regimes.
    \item Implemented dual \textbf{LSTM networks} to capture temporal patterns in cognitive assessments and biomarker sequences, enabling robust sequential modelling.
    \item Engineered a \textbf{dynamic Gated Fusion mechanism} to adaptively weigh contributions from each modality, producing patient-specific, synergistic representations.
    \item Achieved strong diagnostic performance on a three-class classification (CN, MCI, Dementia), with Accuracy = 89.66\%, AUC-ROC = 0.9611, MCC = 0.8337, and G-Mean = 0.9129, demonstrating balanced classification even under severe class imbalance.
    \end{itemize}

\textit{Technologies: Python, PyTorch, ANTsPy, NiBabel, scikit-learn, Google Cloud, ADNI Dataset}
